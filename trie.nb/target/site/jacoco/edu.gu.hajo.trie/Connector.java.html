package edu.gu.hajo.trie;

/**
 *
 * @author Mikael
 */
public final class Connector {
    
    private final TrieNode node;
    
    public Connector(TrieNode node){
        this.node = node;
    }
    
    public void connect(Connector connector){
        node.addPeer(connector.node);
    }
}
